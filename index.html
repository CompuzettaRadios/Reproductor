renderHistory() { const historyList = document.getElementById('historyList'); // Verificar si el elemento existe (ya que ahora usamos iframe) if (!historyList) { console.log('Historial manejado por iframe historial.html'); return; } if (this.historyData.length === 0) { historyList.innerHTML = `
${this.getCurrentTime()}
Música Cristiana 24/7
Stereo Revelación Radio
`; return; } historyList.innerHTML = this.historyData.map(item => { const parts = item.song.split(' - '); const artist = parts[0] || 'Stereo Revelación Radio'; const song = parts[1] || parts[0] || 'Música Cristiana'; return `
${item.time}
${song}
${artist}
`; }).join(''); }
